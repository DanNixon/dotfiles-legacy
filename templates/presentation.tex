\documentclass[xetex,serif,t]{beamer}

% Environment for a slide
\newenvironment{slide}[1]
{
  \begin{frame}
    \frametitle{#1}
    \section{#1}
}
{
  \end{frame}
}

% Environment for a sub slide
\newenvironment{subslide}[1]
{
  \begin{frame}
    \frametitle{#1}
    \subsection{#1}
}
{
  \end{frame}
}

% Environment for a sub sub slide
\newenvironment{subsubslide}[1]
{
  \begin{frame}
    \frametitle{#1}
    \subsubsection{#1}
}
{
  \end{frame}
}
